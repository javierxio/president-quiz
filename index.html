let currentQuestion = 0;

let answers = new Array(10).fill(null);

let scores = {A:0, B:0, C:0, D:0, E:0, F:0, G:0, H:0};


function loadQuestion() {

const q = questions[currentQuestion];

document.getElementById('question-text').textContent = `${currentQuestion + 1}. ${q.q}`;

document.getElementById('question-number').textContent = `Question ${currentQuestion + 1} of 10`;

const optionsDiv = document.getElementById('options');

optionsDiv.innerHTML = '';

optionsText[currentQuestion].forEach((text, i) => {

const letter = String.fromCharCode(65 + i);

const div = document.createElement('div');

div.className = 'option';

div.textContent = text;

div.onclick = () => selectOption(letter, div);

optionsDiv.appendChild(div);

});

document.getElementById('next-btn').disabled = true;

updateProgress();

}


function selectOption(letter, element) {

document.querySelectorAll('.option').forEach(opt => opt.classList.remove('selected'));

element.classList.add('selected');

answers[currentQuestion] = letter;

document.getElementById('next-btn').disabled = false;

}


function updateProgress() {

const percent = ((currentQuestion) / 10) * 100;

document.getElementById('progress-bar').style.width = percent + '%';

}


document.getElementById('next-btn').addEventListener('click', () => {

if (answers[currentQuestion]) {

currentQuestion++;

if (currentQuestion < 10) {

loadQuestion();

} else {

showResults();

}

}

});


function showResults() {

document.getElementById('quiz-screen').style.display = 'none';

document.getElementById('result-screen').style.display = 'block';

// Calculate scores

scores = {A:0, B:0, C:0, D:0, E:0, F:0, G:0, H:0};

answers.forEach(letter => {

if (letter) scores[letter]++;

});

// Find top matches

const sorted = Object.entries(scores).sort((a,b) => b[1] - a[1]);

const topLetter = sorted[0][0];

const topPres = presidents[topLetter];

let html = `

<div style="font-size: 4.5rem; margin: 20px 0;">🇺🇸</div>

<div class="president-name">${topPres.name}</div>

<p class="score">You scored ${sorted[0][1]} / 10 matches</p>

<p style="margin: 25px 0; font-size: 1.15rem; max-width: 500px; margin-left: auto; margin-right: auto;">${topPres.desc}</p>

`;

// Show top 3

if (sorted.length > 1) {

html += `<p style="margin-top:30px; color:#64748b;">Runner-ups:</p><ul style="text-align:left; max-width:420px; margin:15px auto;">`;

for (let i = 1; i < Math.min(3, sorted.length); i++) {

const lettr = sorted[i][0];

html += `<li>${presidents[lettr].name} — ${sorted[i][1]} points</li>`;

}

html += `</ul>`;

}

document.getElementById('result-content').innerHTML = html;

}


function restartQuiz() {

}
